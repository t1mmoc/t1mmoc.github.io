                    var h = parseInt($h[0].tagName.substring(1));
                    var ph = parseInt($ph[0].tagName.substring(1));

                    var offset = parseInt($ph.attr("offset")) - (ph - h);
                    if (offset < 1) {
                        offset = 1;

                    $h.attr("offset", offset);
                } //和上一个h标签层级相等时, 级数不变

                    $h.attr("offset", $ph.attr("offset"));

                    '<li class="scroller-offset' +
                    $h.attr("offset") +
                    '"><a href="#' +
                    $h.attr("id") +
                    '" title="' + $h.text() + '">' +
                    $h.text() +
                    "</a></li>"
                );
